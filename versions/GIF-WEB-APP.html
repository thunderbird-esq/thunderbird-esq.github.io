<!DOCTYPE html>

<html lang="en">

<head>

  <meta charset="UTF-8">

  <title>Image Sequence to GIF</title>

  <style>

    body {

      font-family: Arial, sans-serif;

      margin: 20px;

    }

    #preview {

      display: flex;

      flex-wrap: wrap;

      gap: 10px;

      margin-top: 20px;

    }

    .preview-img {

      object-fit: cover;

      border: 1px solid #ccc;

    }

  </style>

</head>

<body>

  <h2>Image Sequence to GIF</h2>

  <input type="file" id="imageInput" multiple accept="image/*">

  <br><br>

  <label for="cellSize">Cell Size (px):</label>

  <input type="number" id="cellSize" value="100">

  <br><br>

  <button onclick="createGIF()">Create GIF</button>

  <div id="preview"></div>

  <script src="https://cdn.jsdelivr.net/npm/gif.js.optimized/dist/gif.js"></script>

  <script>

    const imageInput = document.getElementById('imageInput');

    const preview = document.getElementById('preview');

    const cellSizeInput = document.getElementById('cellSize');

    let images = [];


    imageInput.addEventListener('change', function() {

      images = [];

      preview.innerHTML = '';

      const files = Array.from(this.files);


      files.forEach(file => {

        const reader = new FileReader();

        reader.onload = function(e) {

          const img = new Image();

          img.src = e.target.result;

          img.classList.add('preview-img');

          img.style.width = cellSizeInput.value + 'px';

          img.style.height = cellSizeInput.value + 'px';

          preview.appendChild(img);

          images.push(img);

        };

        reader.readAsDataURL(file);

      });

    });


    cellSizeInput.addEventListener('input', function() {

      const size = this.value + 'px';

      document.querySelectorAll('.preview-img').forEach(img => {

        img.style.width = size;

        img.style.height = size;

      });

    });


    function createGIF() {

      if (images.length === 0) {

        alert('Please upload images first.');

        return;

      }


      const cellSize = parseInt(cellSizeInput.value, 10);

      const gif = new GIF({

        workers: 2,

        quality: 10,

        width: cellSize,

        height: cellSize

      });


      images.forEach(img => {

        const canvas = document.createElement('canvas');

        canvas.width = cellSize;

        canvas.height = cellSize;

        const ctx = canvas.getContext('2d');

        ctx.drawImage(img, 0, 0, cellSize, cellSize);

        gif.addFrame(canvas, { delay: 500 });

      });


      gif.on('finished', function(blob) {

        const url = URL.createObjectURL(blob);

        const gifImg = document.createElement('img');

        gifImg.src = url;

        preview.innerHTML = '';

        preview.appendChild(gifImg);

      });


      gif.render();

    }

  </script>

</body>

</html>
